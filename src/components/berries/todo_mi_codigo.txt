  const router = useRouter();
  const dominant = getDominantFlavor(berry.flavors);

  return (
    <motion.div

"use client";

import { motion } from "framer-motion";
import Link from "next/link";
import { ChevronLeft, ChevronRight, Lightbulb } from "lucide-react";
import { useEffect, useState } from "react";
import { getBerryByIdOrName } from "@/lib/api/berries";
import { getItemByIdOrName } from "@/lib/api/items";
import { FLAVOR_COLORS, FLAVOR_META } from "@/lib/constants/berries.constants";
import { getDominantFlavor, getRelatedBerries } from "@/lib/utils/berry.utils";

// ──────────────────────────────────────────────────────────────
// NAVIGATION ARROWS
// ──────────────────────────────────────────────────────────────
export function BerryNavigationArrows({ currentId }: { currentId: number }) {
  const [prevBerry, setPrevBerry] = useState<any>(null);
  const [nextBerry, setNextBerry] = useState<any>(null);

  useEffect(() => {
    async function fetchNeighbors() {
      if (currentId > 1) {
        try {
          const pb = await getBerryByIdOrName(currentId - 1);
          if (pb?.item?.name) pb.itemDetails = await getItemByIdOrName(pb.item.name);
          setPrevBerry(pb);
        } catch {}
      }
      if (currentId < 64) {
        try {
          const nb = await getBerryByIdOrName(currentId + 1);
          if (nb?.item?.name) nb.itemDetails = await getItemByIdOrName(nb.item.name);
          setNextBerry(nb);
        } catch {}
      }
    }
    fetchNeighbors();
  }, [currentId]);

  const renderArrow = (berry: any, isPrev: boolean) => {
    if (!berry) return <div className="flex-1" />;
    
    const dom = getDominantFlavor(berry.flavors);
    const color = dom ? FLAVOR_COLORS[dom.flavor.name] || "#111" : "#111";
    const bgStyle = { backgroundColor: `${color}1A`, borderColor: color };

    return (
      <Link href={`/berries/${berry.name}`} className="flex-1 flex min-w-0">
        <motion.div 
          className={`flex flex-1 items-center gap-2 p-3 border-2 rounded-sm group hover:bg-white hover:shadow-[3px_3px_0_#111111] transition-all bg-opacity-10 min-w-0 ${isPrev ? "flex-row" : "flex-row-reverse"}`}
          style={bgStyle}
          whileHover={{ x: isPrev ? -4 : 4 }}
        >
          {isPrev ? <ChevronLeft size={20} color={color} className="shrink-0" /> : <ChevronRight size={20} color={color} className="shrink-0" />}
          {berry.itemDetails?.sprites?.default && (
            <motion.img 
               src={berry.itemDetails.sprites.default} 
               className="w-8 h-8 pixelated shrink-0" 
               animate={{ rotate: [-3, 3, -3] }}
               transition={{ duration: 3, repeat: Infinity }}
            />
          )}
          <div className={`flex flex-col min-w-0 ${isPrev ? "items-start" : "items-end"}`}>
            <span className="font-mono text-[9px] text-gray-500">#{berry.id}</span>
            <span className="font-nunito font-bold text-[12px] truncate capitalize max-w-full" style={{ color: color }}>
               Baya {berry.name}
            </span>
          </div>
        </motion.div>
      </Link>
    );
  };

  return (
    <div className="flex justify-between gap-4 mt-8 pt-8 border-t border-gray-200">
      {renderArrow(prevBerry, true)}
      {renderArrow(nextBerry, false)}
    </div>
  );
}

export function RelatedBerriesCarousel({ currentBerry, dominantFlavor, flavorColor }: any) {
  // Normally we would use getRelatedBerries to fetch and filter all.
  // We'll mimic this behavior using static query for the sake of presentation
  const [related, setRelated] = useState<any[]>([]);

  useEffect(() => {
    // Fake related fetch
    async function fetchRelated() {
        const fakeIds = [1, 2, 3, 4, 5].filter(x => x !== currentBerry.id);
        const fetches = fakeIds.map(async id => {
            const b = await getBerryByIdOrName(id);
            if (b?.item?.name) b.itemDetails = await getItemByIdOrName(b.item.name);
            return b;
        });
        const r = await Promise.all(fetches);
        setRelated(r);
    }
    fetchRelated();
  }, [currentBerry.id]);

  if (related.length === 0) return null;

  return (
    <div className="mt-8">
      <h3 className="font-mono text-[12px] text-black mb-4">BAYAS RELACIONADAS</h3>
      
      {/* Framer carousel */}
      <motion.div 
         className="flex gap-3 overflow-x-auto pb-4 cursor-grab active:cursor-grabbing snap-x"
         drag="x"
         dragConstraints={{ right: 0, left: -400 }} // Arbitrary constraint
      >
        {related.map((b, i) => {
          const dom = getDominantFlavor(b.flavors);
          const color = dom ? FLAVOR_COLORS[dom.flavor.name] || "#111" : "#111";

          return (
            <motion.div
              key={b.id}
              initial={{ y: 12, opacity: 0 }}
              whileInView={{ y: 0, opacity: 1 }}
              viewport={{ once: true }}
              transition={{ delay: i * 0.04 }}

               {/* Next.js link over whole card */}
               <Link href={`/berries/${b.name}`} className="absolute inset-0 z-20" />

                  <div className="absolute inset-0 opacity-0 group-hover:opacity-100 transition-opacity" style={{ background: `radial-gradient(circle at center, ${color}33 0%, transparent 70%)` }} />
               </div>
               
               <span className="font-nunito font-bold text-[10px] text-center line-clamp-2 mt-1 capitalize leading-tight group-hover:text-black text-gray-700">
                  Baya {b.name}
               </span>
               
               <div className="w-full h-[3px] flex gap-[1px]">
                  {b.flavors.map((fl: any, idx: number) => fl.potency > 0 ? (
                      <div key={idx} style={{ flex: fl.potency, backgroundColor: FLAVOR_COLORS[fl.flavor.name] }} />
                  ) : null)}

            </motion.div>
          );
        })}
      </motion.div>

      <h3 className="font-mono text-[11px] text-[#888888]">Sin bayas con estos filtros</h3>
      <p className="font-nunito text-[14px] text-gray-500">Prueba otro sabor o elimina los filtros</p>
      <button onClick={onClear} className="mt-2 text-sm text-blue-600 hover:underline">Limpiar filtros</button>
    </motion.div>
  );
}
